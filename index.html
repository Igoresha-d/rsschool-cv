                        import sys
                        import pygame
                        from pygame.locals import *
                        pygame.init()
                        surface = pygame.display.set_mode((640, 480))
                        clock = pygame.time.Clock()
                        surfrect = surface.get_rect()
                        cat=pygame.image.load("cat.jpg")
                        rect = cat.get_rect()
                        rect.center = (surfrect.w / 2, surfrect.h / 2)
                        touched = False

                        width = surface.get_width()
                        height = surface.get_height()
                        sp0=8
                        sp1=8
                        speed=[sp0,sp1]
                        while True:
                            for ev in pygame.event.get():
                                if ev.type == QUIT:
                                    pygame.quit()
                                elif ev.type == pygame.MOUSEBUTTONDOWN:
                            if rect.collidepoint(ev.pos):
                                touched = True
                                # This is the starting point
                                pygame.mouse.get_rel()
                            elif ev.type == pygame.MOUSEBUTTONUP:
                                touched = False
                                clock.tick(60)
                                surface.fill((0, 0, 0))
                                rect = rect.move(speed)
                            if rect.left < 0 or rect.right > width:
                                speed[0] = -speed[0]
                                sp0=speed[0]
                                surface.fill((255,41,41))
                            if rect.top < 0 or rect.bottom > height:
                                speed[1] = -speed[1]
                                sp1=speed[1]
                                surface.fill((56,197,240))
                            if touched:
                                rect.move_ip(pygame.mouse.get_rel())
                                rect.clamp_ip(surfrect)
                                speed[0]=0
                                speed[1]=0
                                cat=pygame.image.load("cat2.jpg")
                            if not touched:
                                speed[0]= sp0
                                speed[1]= sp1
                                cat=pygame.image.load("cat.jpg")
                            surface.blit(cat,rect)
                            pygame.display.flip()
